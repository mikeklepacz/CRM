A minimal version:

export function planEligibleSchedule({ candidates, batchSize }: PlannerOptions): PlannerResult {
  // group by timezone
  const byTZ = new Map<string, Candidate[]>();

  for (const c of candidates) {
    const tz = c.timezone || 'America/New_York';
    if (!byTZ.has(tz)) byTZ.set(tz, []);
    byTZ.get(tz)!.push(c);
  }

  // quota = proportional to candidate count
  const zones = Array.from(byTZ.entries()).map(([tz, list]) => ({
    timezone: tz,
    candidates: list,
    quota: 0
  }));

  const total = candidates.length;
  let remaining = batchSize;

  for (let i = 0; i < zones.length; i++) {
    const z = zones[i];
    if (i === zones.length - 1) {
      z.quota = Math.min(z.candidates.length, remaining);
    } else {
      z.quota = Math.min(
        Math.round((z.candidates.length / total) * batchSize),
        z.candidates.length
      );
    }
    remaining -= z.quota;
  }

  // round-robin selection
  const balanced: Candidate[] = [];
  let index = 0;

  while (balanced.length < batchSize && zones.some(z => z.quota > 0)) {
    const z = zones[index];
    if (z.quota > 0 && z.candidates.length > 0) {
      balanced.push(z.candidates.shift()!);
      z.quota--;
    }
    index = (index + 1) % zones.length;
  }

  return {
    balancedCandidates: balanced,
    timezonePlans: zones,
    totalEligible: candidates.length,
    totalSchedulable: balanced.length
  };
}
